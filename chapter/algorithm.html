1. 搜索

在我们遇到的一些问题当中，有些问题我们不能够确切的找出数学模型，即找不出一种直接求解的方法，解决这一类问题，我们一般采用搜索的方法解决。搜索就是用问题的所有可能去试探，按照一定的顺序、规则，不断去试探，直到找到问题的解，试完了也没有找到解，那就是无解，试探时一定要试探完所有的情况（实际上就是穷举）；

对于问题的第一个状态，叫==初始状态==，要求的状态叫==目标状态==。搜索就是把规则应用于初始状态，在其产生的状态中，直到得到一个目标状态为止。产生新的状态的过程叫扩展（由一个状态，应用规则，产生新状态的过程）

搜索的要点：

（1）初始状态；

（2）重复产生新状态；

（3）检查新状态是否为目标，是结束，否转（2）；

如果搜索是以接近起始状态的程序依次扩展状态的，叫==宽度优先搜索==。

如果扩展是首先扩展新产生的状态，则叫==深度优先搜索。==

1.1. 深度优先搜索(DFS)

模版：

dfs (当前元素) {
      if (满足结束条件) 结束;
      if (当前元素满足条件) {//或者if(!当前元素不满足条件) return
      改变状态;                        
      dfs(下一个元素);
      回溯状态;
    }
}

1.1.1. 1. LeetCode 17 Letter Combinations of a Phone Number

class Solution {
    String[] strings = {"abc", "def", "ghi", "jkl", "mno", "pqrs", "tuv", "wxyz"};
    public List<String> letterCombinations(String digits) {
        List<String> res = new ArrayList<>();
        if (digits == null || digits.length() == 0) return res; 
        dfs(res, digits.toCharArray(), new StringBuilder(""), 0);
        return res;
    }

    private void dfs(List<String> res, char[] digits, StringBuilder temp, int index) {
        if (index == digits.length) {//满足结束条件
            res.add(temp.toString());
            return;
        }
        char[] chars = strings[digits[index] - '2'].toCharArray();
        for (int i = 0; i < chars.length; i++) {//当前元素满足条件
            temp.append(chars[i]);//改变状态
            dfs(res, digits, temp, index + 1);//dfs下一个元素
            temp.deleteCharAt(temp.length() - 1);//回溯状态
        }
    }
}

1.1.2. 2. LeetCode 79 Word Search

class Solution {
    private int rows, cols;
    public boolean exist(char[][] board, String word) {
        if (board == null || board.length == 0) {
            return false;
        }
        rows = board.length;
        cols = board[0].length;
        boolean[][] visited = new boolean[rows][cols];
        for (int i = 0; i < rows; i++) {
            for (int j = 0; j < cols; j++) {
                if (dfs(board, word.toCharArray(), visited, 0, i, j)) {
                    return true;
                }
            }
        }
        return false;
    }

    private boolean dfs(char[][] board, char[] word, boolean[][] visited, int index, int i, int j) {
        if (index == word.length) return true;//满足结束条件
        if (i >= 0 && i < rows && j >= 0 && j < cols && !visited[i][j] && board[i][j] == word[index]) {//当前元素满足条件
            visited[i][j] = true;//改变状态
            int[] dx = {1, -1, 0, 0}, dy = {0, 0, 1, -1};
            for (int k = 0; k < 4; k++) {
                int x = i + dx[k], y = j + dy[k];
                if (dfs(board, word, visited, index + 1, x, y)) {//dfs下一个元素
                    return true;
                }
            }
            visited[i][j] = false;//回溯状态    
        }
        return false;
    }
}

1.1.3. 3. LeetCode 46 Permutations

class Solution {
    public List<List<Integer>> permute(int[] nums) {
        List<List<Integer>> res = new ArrayList<>();
        boolean[] visited = new boolean[nums.length];
        dfs(nums, visited, res, new Stack<Integer>());
        return res;
    }

    private void dfs(int[] nums, boolean[] visited, List<List<Integer>> res, Stack<Integer> stack) {
        if (stack.size() == nums.length) {//满足结束条件
            res.add(new ArrayList<>(stack));
            return;
        }
        for (int i = 0; i < nums.length; i++) {
            if (!visited[i]) {//当前元素满足条件
                stack.push(nums[i]);
                visited[i] = true;//改变状态
                dfs(nums, visited, res, stack);//回溯下一个元素
                visited[i] = false;//回溯状态
                stack.pop();
            }
        }
    }
}

1.1.4. 4. LeetCode 47 Permutations II

class Solution {
    public List<List<Integer>> permuteUnique(int[] nums) {
        List<List<Integer>> res = new ArrayList<>();
        boolean[] visited = new boolean[nums.length];
        Arrays.sort(nums);
        dfs(nums, res, new Stack<Integer>(), visited);
        return res;
    }

    private void dfs(int[] nums, List<List<Integer>> res, Stack<Integer> stack, boolean[] visited) {
        if (stack.size() == nums.length) {
            res.add(new ArrayList<>(stack));
            return;
        }
        for (int i = 0; i < nums.length; i++) {
            if (!visited[i]) {
                if (i > 0 && nums[i] == nums[i - 1] && !visited[i - 1]) continue;
                stack.push(nums[i]);
                visited[i] = true;
                dfs(nums, res, stack, visited);
                visited[i] = false;
                stack.pop();
            }
        }
    }
}

1.1.7. 7. LeetCode 216 Combination Sum III

class Solution {
    public List<List<Integer>> combinationSum3(int k, int n) {
        List<List<Integer>> res = new ArrayList<>();
        dfs(k, n, res, new Stack<Integer>(), 1);
        return res;
    }

    private void dfs(int k, int n, List<List<Integer>> res, Stack<Integer> stack, int curr) {
        if (k == stack.size() && n == 0) {
            res.add(new ArrayList(stack));
        }
        if (n >= curr) {
            for (int i = curr; i <= 9; i++) {
                stack.push(i);
                dfs(k, n - i, res, stack, i + 1);
                stack.pop();
            }
        }
    }
}

1.1.8. 8. LeetCode 52 N-Queens II

class Solution {
    int res = 0, N = 0;
    public int totalNQueens(int n) {
        if (n == 0) {
            return 0;
        }
        N = n;
        boolean[] c = new boolean[n], l_down = new boolean[n * 2 - 1], l_up = new boolean[n * 2 - 1];
        dfs(0, c, l_down, l_up);
        return res;
    }

    private void dfs(int i, boolean[] c, boolean[] l_down, boolean[] l_up) {
        if (i == N) {
            res++;
            return;
        }
        for (int j = 0; j < N; j++) {
            if (!c[j] && !l_down[i + j] && !l_up[i - j + N - 1]) {
                c[j] = true;
                l_down[i + j] = true;
                l_up[i - j + N - 1] = true;
                dfs(i + 1, c, l_down, l_up);
                c[j] = false;
                l_down[i + j] = false;
                l_up[i - j + N - 1] = false;
            }
        }
    }
}

注意：主函数第9行调用dfs，如果只有1棵递归树，只调用1次。若有多棵递归树，如LeetCode79，则for内调用。

1.1.9. 9. LeetCode 37 Sudoku Solver

class Solution {
    public void solveSudoku(char[][] board) {
        boolean[][] srow=new boolean[9][9],scol=new boolean[9][9],st=new boolean[9][9];
        for (int i = 0; i < 9; i++) {
            for (int j = 0; j < 9; j++) {
                if(board[i][j]!='.'){
                    char ch=board[i][j];
                    srow[i][ch-'0'-1]=true;
                    scol[j][ch-'0'-1]=true;
                    st[i/3*3+j/3][ch-'0'-1]=true;
                }
            }
        }
        dfs(board,0,0,srow,scol,st);
    }

    private boolean dfs(char[][] board,int r,int c, boolean[][] srow, boolean[][] scol, boolean[][] st) {
        if(c==9){
            r++;
            c=0;
        }
        if(r==9){
            return true;
        }
        if(board[r][c]!='.'){
            return dfs(board,r,c+1,srow,scol,st);
        }
        for (int k = 1; k <=9; k++) {
            if(!srow[r][k-1]&&!scol[c][k-1]&&!st[r/3*3+c/3][k-1]){
                board[r][c]=(char)(k+'0');
                srow[r][k-1]=true;
                scol[c][k-1]=true;
                st[r/3*3+c/3][k-1]=true;
                if(dfs(board,r,c+1,srow,scol,st)){
                    return true;
                }
                else {
                    board[r][c]='.';
                    srow[r][k-1]=false;
                    scol[c][k-1]=false;
                    st[r/3*3+c/3][k-1]=false;
                }
            }
        }
        return false;
    }
}

注意：如果结果是递归树在某一条件的元素或性质，需要返回boolean类型判断是否找到，并只在false时回溯，true时直接返回则保存当前的状态，如LeetCode 79、37。如果结果是由递归树在某一条件的元素的副本组成的集合，则可以用void返回，因为只要知道状态即可将其副本存入结果集，回溯也不会影响结果集。

2.1. 双指针之滑动窗口

问题描述：在一个序列中寻找满足某个条件的子序列，返回这些子序列的某性质，该性质需满足以下条件：当【l，r】表示满足性质的子序列时，【l-，r+】亦满足；【l，r】表示不满足性质的子序列时，【l+，r-】亦不满足。（r+表示r+1、r+2...；l+表示l+1、l+2...）

模版代码：

int left = 0, right = 0;
while (right < s.size()) {
    window.add(s[right]);      //增大当前窗口
    right++;                   //r指针向右移动
    while (check(window)){     //满足某性质的所有窗口
        window.remove(s[left]);//减小当前窗口
        left++;                              //l指针向右移动 
    }
      //不满足某性质的所有窗口
}

说明：此模版迭代了所有可能的窗口，答案通常是满足（或不满足）某性质（check(window)）的所有窗口的某一性质（如窗口最短长度），需要在第5行while中或第9行维护所求性质。代码中先增大窗口（第3行）再进行判断（第5行）保证了不遗漏情况，写代码时我们要保持先“处理”，后“判断”的顺序，保证整个逻辑更清晰。

为了便于描述，后面使用"window"表示窗口集合，使用"check(window)"表示窗口是否满足某一性质，使用"satisfied(window)"表示满足性质的窗口集合，使用"unsatisfied(window)"表示不满足性质的窗口集合，使用"len"表示窗口长度，使用"condition -> window"表示某条件下对应的窗口，使用"answer"表示最终答案。

2.1.1. 1. LeetCode 209 Minimum Size Subarray Sum

check(window)：sum(window) >= s

answer：min(len(satisfied(window)))

//长度最小的子数组: 找出数组中满足其和≥s的长度最小的连续子数组,不存在返回0。
//输入：s = 7, nums = [2,3,1,2,4,3] 输出: 2    (4,3)
public int minSubArrayLen(int s, int[] nums) {
    int l = 0, r = 0, minLen = Integer.MAX_VALUE, sum = 0;
    while (r < nums.length) {
      sum += nums[r++];                        //增大当前窗口,r指针右移
      while (sum >= s) {                        //满足性质（和>=s）
        minLen = Math.min(minLen, r - l);//维护所求性质（长度最小）
        sum -= nums[l++];
      }
    }
    return minLen == Integer.MAX_VALUE ? 0 : minLen;
}

2.1.2. 2. LeetCode 3 Longest Substring Without Repeating Characters

check(window)：window第r个字符是否重复

answer：max(len(unsatisfied(window)))

//无重复字符的最长子串: 找出其中不含有重复字符的最长子串的长度。
//输入: "pwwkew" 输出: 3    (wke)
public int lengthOfLongestSubstring(String s) {
    int l = 0, r = 0, maxLen = 0;
    int[] map = new int[256];
    while (r < s.length()) {
      map[s.charAt(r++)]++;
      while (map[s.charAt(r - 1)] != 1) {
        map[s.charAt(l++)]--;
      }
      maxLen = Math.max(maxLen, r - l);
    }
    return maxLen;
}

说明：由于ascII码可以表示256个字符，可以用 new int[256] 来维护一个map，用于记录每个字符出现的次数。增大窗口后，map[s.charAt(r - 1)] == 1 表示第r个字符只出现1次，无重复。

2.1.3. 3. LeetCode 76 Minimum Window Substring

check(window)：window中是否包含t中全部字符

answer：min(len(satisfied(window))) -> window

//最小覆盖子串: 在字符串S里面找出包含T所有字母的最小子串。
//输入: S = "ADOBECODEBANC", T = "ABC" 输出: "BANC"
public String minWindow(String s, String t) {
    int l = 0, r = 0, minLen = Integer.MAX_VALUE, first = -1, count = 0, now = 0;
    int[] maps = new int[256], mapt = new int[256];
    for (int i = 0; i < t.length(); i++) {
      mapt[t.charAt(i)]++;
      if (mapt[t.charAt(i)] == 1) {
        count++;//count表示t中需要匹配字符的种类数(aabb则为2种，为a和b)
      }
    }
    while (r < s.length()) {
      maps[s.charAt(r)]++;
      if (maps[s.charAt(r)] == mapt[s.charAt(r++)]) {
        now++;//now表示window已匹配t中字符种类数
      }
      while (now == count) {
        if (r - l < minLen) {
          minLen = r - l;
          first = l;
        }
        maps[s.charAt(l)]--;
        if (maps[s.charAt(l)] == mapt[s.charAt(l++)] - 1) {
          now--;
        }
      }
    }
    return first == -1 ? "" : s.substring(first, first + minLen);
}

2.1.4. 4. LeetCode 340 Longest Substring with At Most K Distinct Characters

​ LintCode 384 Longest Substring with At Most K Distinct Characters

check(window)：window包含不同字符个数>k

answer：max(len(unsatisfied(window)))

//最多有k个不同字符的最长子字符串: 给定字符串S，找到最多有k个不同字符的最长子串T。
//输入: S = "eceba" 并且 k = 3 输出: 4    (eceb)
public int lengthOfLongestSubstringKDistinct(String s, int k) {
    int[] map = new int[256];
    int l = 0, r = 0, now = 0, maxLen = 0;
    while (r < s.length()) {
      map[s.charAt(r)]++;
      if (map[s.charAt(r++)] == 1) {
        now++;
      }
      while (now > k) {
        map[s.charAt(l)]--;
        if (map[s.charAt(l++)] == 0) {
          now--;
        }
      }
      maxLen = Math.max(maxLen, r - l);
    }
    return maxLen;
}

2.1.5. 5. LeetCode 424 Longest Repeating Character Replacement

check(window)： maxCount + k < r - l

answer：max(len(unsatisfied(window)))

//替换后的最长重复字符: 最多替换k次，找到包含重复字母的最长子串的长度。
//输入: s = "AABABBA", k = 1 输出: 4    (AABA)
public int characterReplacement(String s, int k) {
    int l = 0, r = 0, maxCount = 0, maxLen = 0;
    int[] map = new int[26];
    while (r < s.length()) {
      map[s.charAt(r) - 'A']++;
      if (map[s.charAt(r) - 'A'] > maxCount) {
        maxCount = map[s.charAt(r) - 'A'];
      }
      r++;
      while (maxCount + k < r - l) {
        map[s.charAt(l++) - 'A']--;          
      }
      maxLen = Math.max(maxLen, r - l);
    }
    return maxLen; 
}

说明：1.第10行的while可以换为if，由于题目所求的是最大窗口长度(maxLen)，并且小于maxLen解的求解过程（缩小窗口）对后面超过maxLen解的求解过程（增大窗口）不产生影响，故无需求小于maxLen的解。如：ABCABBA，k = 2，当前窗口为ABCAB时不满足条件(maxCount+k<r-l)，此时maxLen=4，对应窗口ABCA，则l只需右移一次更新窗口起点，无需求<maxLen窗口(CAB、AB、B)。若小于maxLen解的求解过程对后面有影响，则需求所有窗口。如上第2题 LeetCode 3，pwwkew，当前窗口为pww时，不满足条件，但后面判断窗口是否重复时要保证当前元素之前是无重复的，故l需右移到第一个无重复窗口。 2.maxCount记录的是元素出现次数的历史最大值，因为使maxLen发生变化的是历史最大值，不是当前最大值，而当窗口缩小时，当前最大值发生变化，但maxLen不会改变。

2.1.6. 6. LeetCode 438 Find All Anagrams in a String

check(window)： window中是否包含p中全部字符

answer：len(unsatisfied(window)) == p -> l

//找到字符串中所有字母异位词: 找到 s 中所有是 p 的字母异位词的子串，返回这些子串的起始索引。
//输入: s: "cbaebabacd" p: "abc" 输出: [0, 6]
public List<Integer> findAnagrams(String s, String p) {
    int l = 0, r = 0, now = 0, k = 0;
    List<Integer> res = new ArrayList<>();
    int[] maps = new int[26], mapp = new int[26];
    for (int i = 0; i < p.length(); i++) {
      mapp[p.charAt(i) - 'a']++;
      if (mapp[p.charAt(i) - 'a'] == 1) {
        k++;
      }
    }
    while (r < s.length()) {
      maps[s.charAt(r) - 'a']++;
      if (mapp[s.charAt(r) - 'a'] == maps[s.charAt(r) - 'a']) {
        now++;
      }
      r++;
      while (now == k) {
        if (p.length() == r - l) {
          res.add(l);
        }
        maps[s.charAt(l) - 'a']--;
        if (maps[s.charAt(l) - 'a'] == mapp[s.charAt(l) - 'a'] - 1) {
          now--;
        }
        l++;
      }
    }
    return res;
}

2.1.7. 7. LeetCode 567 Permutation in String

//字符串的排列: 给定两个字符串 s1 和 s2，写一个函数来判断 s2 是否包含 s1 的排列。
//输入: s1 = "ab" s2 = "eidbaooo" 输出: True
public boolean checkInclusion(String p, String s) {
    int l = 0, r = 0, now = 0, k = 0;
    int[] maps = new int[26], mapp = new int[26];
    for (int i = 0; i < p.length(); i++) {
      mapp[p.charAt(i) - 'a']++;
      if (mapp[p.charAt(i) - 'a'] == 1) {
        k++;
      }
    }
    while (r < s.length()) {
      maps[s.charAt(r) - 'a']++;
      if (mapp[s.charAt(r) - 'a'] == maps[s.charAt(r) - 'a']) {
        now++;
      }
      r++;
      while (now == k) {
        if (p.length() == r - l) {
          return true;
        }
        maps[s.charAt(l) - 'a']--;
        if (maps[s.charAt(l) - 'a'] == mapp[s.charAt(l) - 'a'] - 1) {
          now--;
        }
        l++;
      }
    }
    return false;
}

3.1. 并查集(Union Find)

并查集是一种树型的数据结构，用于处理一些不相交集合的==合并==及==查询==问题。

模版：

class UnionFind {
    private int[] father;
    private int count; //统计集合数(连通度)
    public UnionFind (int n) {
        father = new int[n];
        for (int i = 0; i < n; i++) {
            father[i] = i;
        }
    } 
    int find(int x) { //查询
        if (father[x] == x) {
            return x;
        }
        return father[x] = find(father[x]);
    }
    void connect(int x, int y) { //合并
        int rootx = find(x), rooty = find(y);
        if (rootx != rooty) {
            father[rootx] = rooty;
            count--;
        }
    }
    int getCount() {
        return count;
    }
    void setCount(int count) {
        this.count = count;
    }
}

            }
        }

    }
}

class Solution {
    public void solve(char[][] board) {

            return;
        }
        int[] dx = {1, -1, 0, 0}, dy = {0, 0, 1, -1};
        int rows = board.length,cols = board[0].length;
        UnionFind u = new UnionFind(rows * cols + 1);
        int dummyNode = rows * cols;
        for (int i = 0; i < rows; i++) {
            for (int j = 0; j < cols; j++) {
                if (board[i][j] == 'O') {
                    if ((i == 0 || j == 0 || i == rows - 1 || j == cols - 1)) {
                        u.connect(dummyNode, i * cols + j);
                        continue;
                    }
                    for (int k = 0; k < 4; k++) {
                        int x = i + dx[k], y = j + dy[k];
                        if (x >= 0 && x < rows && y >= 0 && y < cols && board[x][y] == 'O') {
                            u.connect(i * cols + j, x * cols + y);
                        }
                    }
                }
            }
        }
        for (int i = 0; i < rows; i++) {
            for (int j = 0; j < cols; j++) {
                if (u.find(i * cols + j) != u.find(dummyNode)) {
                    board[i][j] = 'X';
                }
            }
        }
    }
}

3.1.6. 6. LeetCode 685 Redundant Connection II

class Solution {
    public int[] findRedundantDirectedConnection(int[][] edges) {
        int[] degree = new int[edges.length + 1];
        for (int[] edge : edges) {
            degree[edge[1]]++;//统计每个点的入度
        }
        for (int i = edges.length - 1; i >= 0; i--) {
            if (degree[edges[i][1]] == 2) {//入度为2，判断删除后是否连通
                if (helper(edges, i)) return edges[i];
            }
        }
        for (int i = edges.length - 1; i >= 0; i--) {
            if (helper(edges, i)) return edges[i];
        }
        return new int[0];
    }
      //因为并查集不支持删除操作，所以去除某边只能尝试不添加该边。
    public boolean helper(int[][] edges, int i) {
        UnionFind u = new UnionFind(edges.length + 1);
        u.setCount(edges.length);
        for (int j = 0; j < edges.length; j++) {
            if (i != j) {
                u.connect(edges[j][0], edges[j][1]);
            }
        }
        return u.getCount() == 1;
    }
}

说明：入度为2代表这个点有两个父节点那么肯定要去点一个，那就==从后往前==看看一旦去除仍可连通就返回。 如果没有入度为2。那就从后往前找到去掉第一个去掉后仍联通的边。

            return res;
        }

        HashMap<String, Integer> map = new HashMap<>();
        for (int i = 0; i < accounts.size(); i++) {//email -> account，connect
            List<String> account = accounts.get(i);
            for (int j = 1; j < account.size(); j++) {
                String email = account.get(j);
                map.putIfAbsent(email, i);//13行等价于14行

                uf.connect(map.get(email), i);
            }
        }
        HashMap<Integer, List<String>> indexEmail = new HashMap<>();
        for (String email : map.keySet()) {//id -> email
            int root = uf.find(map.get(email));

            indexEmail.get(root).add(email);

            Collections.sort(emails);
            account.addAll(emails);
            res.add(account);
        }
        return res;
    }

3.2. 字典树(Trie Tree)

又称单词查找树，Trie树，是一种树形结构，是一种哈希树的变种。典型应用是用于统计，排序和保存大量的字符串（但不仅限于字符串），所以经常被搜索引擎系统用于文本词频统计。它的优点是：利用字符串的公共前缀来减少查询时间，最大限度地减少无谓的字符串比较，查询效率比哈希树高。

模版：

class TrieNode {
    public boolean isWord; 
    public TrieNode[] children = new TrieNode[26];
    public TrieNode() {}
}

public class Trie {
    private TrieNode root;
    public Trie() {
        root = new TrieNode();
    }

    public void insert(String word) {
        TrieNode ws = root;
        for(int i = 0; i < word.length(); i++){
            char c = word.charAt(i);
            if(ws.children[c - 'a'] == null){
                ws.children[c - 'a'] = new TrieNode();
            }
            ws = ws.children[c - 'a'];
        }
        ws.isWord = true;
    }

    public boolean search(String word) {
        TrieNode ws = root; 
        for(int i = 0; i < word.length(); i++){
            char c = word.charAt(i);
            if(ws.children[c - 'a'] == null) return false;
            ws = ws.children[c - 'a'];
        }
        return ws.isWord;
    }

    public boolean startsWith(String prefix) {
        TrieNode ws = root; 
        for(int i = 0; i < prefix.length(); i++){
            char c = prefix.charAt(i);
            if(ws.children[c - 'a'] == null) return false;
            ws = ws.children[c - 'a'];
        }
        return true;
    }
}

3.2.1. 1. LeetCode 211 Add and Search Word - Data structure design

class WordDictionary {
    private TrieNode root;
    public WordDictionary() {
        root = new TrieNode();
    }
    public void addWord(String word) {
        TrieNode ws = root;
        for (int i = 0; i < word.length(); i++) {
            TrieNode child = ws.children[word.charAt(i) - 'a'];
            if (child == null) {
                child = new TrieNode();
            }
            ws = child;
        }
        ws.isWord = true;
    }
    public boolean search(String word) {
        return match1(word.toCharArray(), root, 0);
    }
      //只遇到'.'时递归
    private boolean match1(char[] chars, TrieNode ws, int curr) {
        for (int i = curr; i < chars.length; i++) {
            char c = chars[i];
            if (c == '.') {
                for (int j = 0; j < 26; j++) {
                    if (ws.children[j] != null && match1(chars, ws.children[j], i + 1)) {
                        return true;
                    }
                }
                return false;
            }
            else {
                if (ws.children[c - 'a'] == null) return false;
                ws = ws.children[c - 'a'];
            }
        }
        return ws.isWord;
    }
        //完全递归
    private boolean match(char[] chars, TrieNode ws, int curr) {
        if (curr == chars.length) {
            return ws.isWord;
        }
        if (chars[curr] == '.') {
            for (int i = 0; i < ws.children.length; i++) {
                if (ws.children[i] != null && match(chars, ws.children[i], curr + 1)) {
                    return true;
                }
            }
        } else {
            return ws.children[chars[curr] - 'a'] != null && match(chars, ws.children[chars[curr] - 'a'], curr + 1);
        }
        return false;
    }
}

3.2.2. 2. LeetCode 212 Word Search II

class Solution {
    int rows, cols;
    public List<String> findWords(char[][] board, String[] words) {
        List<String> res = new ArrayList<>();
        Set<String> set = new HashSet<>();
        if (board == null || board.length == 0) {
            return res;
        }
        rows = board.length;
        cols = board[0].length;
        boolean[][] visited = new boolean[rows][cols];
        TrieNode root = new TrieNode();
        for (String word : words) {
            TrieNode ws = root;
            char[] chars = word.toCharArray();
            for (char aChar : chars) {
                if (ws.children[aChar - 'a'] == null) {
                    ws.children[aChar - 'a'] = new TrieNode();
                }
                ws = ws.children[aChar - 'a'];
            }
            ws.isWord = true;
        }
        for (int i = 0; i < rows; i++) {
            for (int j = 0; j < cols; j++) {
                dfs(board, root, visited, set, new StringBuilder(""), i, j);
            }
        }
        res.addAll(set);
        return res;
    }

    private void dfs(char[][] board, TrieNode ws, boolean[][] visited, Set<String> set, StringBuilder temp, int i, int j) {
        if (ws.children[board[i][j] - 'a'] == null) return;
        visited[i][j] = true;
        temp.append(board[i][j]);
        if (ws.children[board[i][j] - 'a'].isWord) {
            set.add(temp.toString());
        }
        int[] dx = {1, -1, 0, 0}, dy = {0, 0, 1, -1};
        for (int k = 0; k < 4; k++) {
            int x = i + dx[k], y = j + dy[k];
            if (x >= 0 && x < rows && y >= 0 && y < cols && !visited[x][y]) {
                dfs(board, ws.children[board[i][j] - 'a'], visited, set, temp, x, y);
            }
        }
        temp.deleteCharAt(temp.length() - 1);
        visited[i][j] = false;
    }
}
